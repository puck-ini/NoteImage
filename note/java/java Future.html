# Future 接口

# Future

Future 是 jdk 中表示异步结果的接口。

Future 接口包含以下方法：

/**
* 尝试取消任务
*/
boolean cancel(boolean mayInterruptIfRunning);

/**
* 如果此任务在正常完成之前被取消，则返回true
*/
boolean isCancelled();

/**
* 如果此任务完成，则返true，任务的完成包括正常完成，出现异常，已经被取消
*/
boolean isDone();

/**
* 获取异步结果，如果为完成会一直等待
*/
V get() throws InterruptedException, ExecutionException;

/**
* 获取异步结果，如果还没执行完成会阻塞等待一定的时间
*/
V get(long timeout, TimeUnit unit)
        throws InterruptedException, ExecutionException, TimeoutException;

# FutureTask

FutureTask 是 Future 的一个实现类。RunnableFuture Runnable Future

// FutureTask 构造方法，创建一个 FutureTask 实例需要传入一个 Callable对象
public FutureTask(Callable<V> callable) {
        if (callable == null)
            throw new NullPointerException();
        this.callable = callable;
        this.state = NEW;
}


// 该方法实现了 Runnable 的 run() 方法，用于异步执行Callable获取结果
public void run() {
        if (state != NEW ||
            !UNSAFE.compareAndSwapObject(this, runnerOffset,
                                         null, Thread.currentThread()))
            return;
        try {
            Callable<V> c = callable;
            if (c != null && state == NEW) {
                V result;
                boolean ran;
                try {
                    // 调用 call() 方法获取返回值
                    result = c.call();
                    ran = true;
                } catch (Throwable ex) {
                    result = null;
                    ran = false;
                    setException(ex);
                }
                if (ran)
                    set(result);
            }
        } finally {
            runner = null;
            int s = state;
            if (s >= INTERRUPTING)
                handlePossibleCancellationInterrupt(s);
        }
}

// 设置结果
protected void set(V v) {
        if (UNSAFE.compareAndSwapInt(this, stateOffset, NEW, COMPLETING)) {
            outcome = v;
            UNSAFE.putOrderedInt(this, stateOffset, NORMAL); // final state
            finishCompletion();
        }
}


// 唤醒阻塞的线程
private void finishCompletion() {
        // assert state > COMPLETING;
        for (WaitNode q; (q = waiters) != null;) {
            if (UNSAFE.compareAndSwapObject(this, waitersOffset, q, null)) {
                for (;;) {
                    Thread t = q.thread;
                    if (t != null) {
                        q.thread = null;
                        LockSupport.unpark(t);
                    }
                    WaitNode next = q.next;
                    if (next == null)
                        break;
                    q.next = null; // unlink to help gc
                    q = next;
                }
                break;
            }
        }

        done();

        callable = null;        // to reduce footprint
}

通过 get() 方法获取结果(阻塞唤醒)：

wait()/notify()、Condition.wait()/Condition.signal()、LockSupport.park()/LockSupport.unpark()

// get() 会调用 awaitDone() 方法阻塞等待
public V get() throws InterruptedException, ExecutionException {
        int s = state;
        if (s <= COMPLETING)
            s = awaitDone(false, 0L);
        return report(s);
}

// LockSupport.parkNanos() LockSupport.park()
private int awaitDone(boolean timed, long nanos)
        throws InterruptedException {
        final long deadline = timed ? System.nanoTime() + nanos : 0L;
        WaitNode q = null;
        boolean queued = false;
        for (;;) {
            if (Thread.interrupted()) {
                removeWaiter(q);
                throw new InterruptedException();
            }

            int s = state;
            if (s > COMPLETING) {
                if (q != null)
                    q.thread = null;
                return s;
            }
            else if (s == COMPLETING) // cannot time out yet
                Thread.yield();
            else if (q == null)
                q = new WaitNode();
            else if (!queued)
                queued = UNSAFE.compareAndSwapObject(this, waitersOffset,
                                                     q.next = waiters, q);
            else if (timed) {
                nanos = deadline - System.nanoTime();
                if (nanos <= 0L) {
                    removeWaiter(q);
                    return state;
                }
                LockSupport.parkNanos(this, nanos);
            }
            else
                LockSupport.park(this);
        }
}
